local function __create_tween(__object : Instance, __properties : { [any]: any }, __duration : number?) : Tween
	local __tween = Service.TweenService:Create(__object, TweenInfo.new(__duration or 0.3, Enum.EasingStyle.Quint), __properties)
	__tween:Play()

	return __tween
end

local function __create_round(__parent : Instance, __radius : number?) : UICorner
	local __corner : UICorner = Instance.new('UICorner')

	__corner.CornerRadius = UDim.new(0, __radius or 8)
	__corner.Parent = __parent

	return __corner
end

function Library:get_device() : ()
	local __device : string = 'unknown'

	if not UserInputService.TouchEnabled then
		__device = 'pc'
	else
		__device = 'mobile'
	end

	self.__device = __device
end

local function __update_notification_positions() : ()
	for __i, __notify in ipairs(__Notifications) do
		local __target_y : number = -20

		for __j = #__Notifications, __i + 1, -1 do
			if not __Notifications[__j] or not __Notifications[__j].AbsoluteSize then
				continue
			end

			__target_y = __target_y - __Notifications[__j].AbsoluteSize.Y - 10
		end

		if not __notify or not __notify.Parent then
			continue
		end

		__create_tween(__notify, {Position = UDim2.new(1, -20, 1, __target_y)}, 0.3)
	end
end

function Library:notify(__index : { title: string, content: string, duration: number?, notify_type: string }) : ()
	local __auto_selection : string = __index.notify_type or 'normal'

	if not __NotifyIcons[__auto_selection] then
		__auto_selection = 'normal'
	end

	table.insert(__NotificationQueue, {
		title = __index.title,
		content = __index.content,
		duration = __index.duration or 3,
		notify_type = __auto_selection
	})

	if #__Notifications >= 5 then
		return
	end

	task.spawn(__notification_init)
end

function Library:get_screen_scale() : ()
	local __viewport_size_x : number = Camera.ViewportSize.X

	self.__ui_scale = __viewport_size_x / 1400
end

function Library:mobile_ui_scale() : ()
	if self.__device ~= 'mobile' and self.__device ~= 'unknown' then
		return
	end

	for _, __object in pairs(self.__window.main:GetDescendants()) do
		if __object:IsA('TextLabel') or __object:IsA('TextButton') or __object:IsA('TextBox') then
			if not __object.TextSize then
				continue
			end

			__object.TextSize = math.max(12, math.floor(__object.TextSize * 1.05))
		elseif __object:IsA('ImageLabel') or __object:IsA('ImageButton') then
			if not (__object.Size.X.Offset > 0 and __object.Size.X.Offset < 30) then
				continue
			end

			__object.Size = UDim2.new(0, math.floor(__object.Size.X.Offset * 0.9), 0, math.floor(__object.Size.Y.Offset * 0.9))
		end
	end
end

function Library:mobile_controls() : ()
	if self.__device ~= 'mobile' then
		return
	end

	for _, __obj in pairs(self.__window.main:GetDescendants()) do
		if not ((__obj:IsA('TextButton') or __obj:IsA('ImageButton')) and __obj.Size.Y.Offset < 40) then
			continue
		end

		local __hitbox : Frame = Instance.new('Frame')
		__hitbox.Name = 'TouchHitbox'
		__hitbox.Size = UDim2.new(1, 10, 1, 10)
		__hitbox.Position = UDim2.new(0.5, 0, 0.5, 0)
		__hitbox.AnchorPoint = Vector2.new(0.5, 0.5)
		__hitbox.BackgroundTransparency = 1
		__hitbox.ZIndex = __obj.ZIndex
		__hitbox.Parent = __obj

		if not __obj:IsA('TextButton') then
			continue
		end

		__hitbox.InputBegan:Connect(function(__input)
			if __input.UserInputType ~= Enum.UserInputType.Touch then
				return
			end

			__obj.MouseButton1Click:Fire()
		end)
	end
end

					__value = math.floor(__min + (__max - __min) * __position)

					if __rounding > 0 then
						__value = math.floor(__value / __rounding) * __rounding
					end

					__value = math.clamp(__value, __min, __max)
					__value_label.Text = tostring(__value)

					__create_tween(__slider_track, {Size = UDim2.new(__position, 0, 1, 0)}, 0.1)

					end

					__value_label.Text = tostring(__value)
					__slider_track.Size = UDim2.new((__value - __min) / (__max - __min), 0, 1, 0)

				end

				end)

						return
					end

					__update_slider(__input)
				end)

				table.insert(__module.__elements, __item_frame)

				if __toggled then
					__update_module_height()
				end

				return {
					Set = function(__self, __new_value : number)
						__set_value(__new_value)
					end
				}
			end

							__create_tween(__option_highlight_effect, {BackgroundTransparency = 1}, 0.3)

				end

					local __search_text : string = __dropdown_search.Text:lower()

					for _, __child in pairs(__options_container:GetChildren()) do

							continue
						end

						local __label_child = __child:FindFirstChildOfClass('TextLabel')

						if __label_child then
							if __search_text == '' then

						return
					end

					__opened = true

						end

						end

								__create_tween(__label_child, {TextTransparency = 0}, 0.2)

				end)

				end

						end
					end
				end)

					Set = function(__self, __option : string | { string })

						else
							__selected = __option :: string
						end

						__item_callback(__selected)
					end
				}
			end

			return __module
		end

local Console : Console = {
	__brand = 'Euphoria',
	__prefix = '[Euphoria]',
	__delay = 0.04,
	__running = false,
	__steps = {},
}
Console.__index = Console

local __SPINNER = { '|', '/', '-', '\\' }

local function __pad2(__n : number) : string
	if __n < 10 then
		return '0' .. tostring(__n)
	end

	return tostring(__n)
end

local function __timestamp() : string
	local __t = os.date('*t')

	return __pad2(__t.hour) .. ':' .. __pad2(__t.min) .. ':' .. __pad2(__t.sec)
end

local function __bar(__percent : number, __width : number?) : string
	local __w = __width or 40
	local __p = math.clamp(math.floor(__percent + 0.5), 0, 100)
	local __filled = math.floor((__p / 100) * __w)

	return string.rep('█', __filled) .. string.rep('░', __w - __filled)
end

function Console:init(__index : { brand: string?, delay: number? }?) : Console
	__index = __index or {}

	self.__brand = __index.brand or 'Euphoria'
	self.__prefix = '[' .. self.__brand .. ']'
	self.__delay = __index.delay or 0.04
	self.__running = false
	self.__steps = {}

	return self
end

function Console:__out(__text : string, __kind : string?) : ()
	__kind = __kind or 'print'

	if __kind == 'warn' then
		warn(__text)
	elseif __kind == 'error' then
		warn('[ERR] ' .. __text)
	else
		print(__text)
	end
end

function Console:line(__text : string) : ()
	self:__out(__text, 'print')
	task.wait(self.__delay)
end

function Console:warn_line(__text : string) : ()
	self:__out(__text, 'warn')
	task.wait(self.__delay)
end

function Console:error_line(__text : string) : ()
	self:__out(__text, 'error')
	task.wait(self.__delay)
end

function Console:blank() : ()
	self:__out('')
	task.wait(self.__delay * 0.5)
end

function Console:header(__lines : { string }) : ()
	for _, __line in ipairs(__lines) do
		if __line == '' then
			self:blank()
		else
			task.wait(self.__delay * 2.2)
			self:__out(__line, 'print')
		end
	end

	self:blank()
end

function Console:status(__message : string, __state : string?) : ()
	local __text = self.__prefix .. ': ' .. __message

	if __state then
		__text ..= ' [' .. __state .. ']'
	end

	self:__out(__text, 'print')
	task.wait(self.__delay * 2)
end

function Console:timestamp_log(__text : string) : ()
	self:__out(__timestamp() .. ' -- ' .. __text, 'print')
	task.wait(self.__delay)
end

function Console:progress(__percent : number) : ()
	local __p = math.clamp(math.floor(__percent + 0.5), 0, 100)
	local __text = self.__prefix .. ': [ ' .. __bar(__p) .. ' ] % ' .. tostring(__p)

	self:__out(__text, 'print')
end

function Console:spinner(__text : string, __duration : number?) : ()
	local __d = __duration or 0.75
	local __t0 = os.clock()
	local __i = 1

	while os.clock() - __t0 < __d do
		self:__out(self.__prefix .. ': ' .. __text .. ' ' .. __SPINNER[__i], 'print')
		__i = (__i % #__SPINNER) + 1
		task.wait(0.08)
	end
end

function Console:add_step(__name : string, __run : () -> (boolean, string?)) : ()
	table.insert(self.__steps, {
		name = __name,
		run = __run,
	})
end

local function __default_steps() : { { name: string, run: () -> (boolean, string?) } }
	local Players = game:GetService('Players')
	local RunService = game:GetService('RunService')

	return {
		{
			name = 'Game loaded',
			run = function()
				if not game:IsLoaded() then
					game.Loaded:Wait()
				end

				return true, 'place is ready'
			end,
		},
		{
			name = 'DataModel services',
			run = function()
				local __need = { 'Players', 'UserInputService', 'TweenService', 'RunService', 'CoreGui', 'Lighting' }

				for _, __name in ipairs(__need) do
					local __ok, __svc = pcall(function()
						return game:GetService(__name)
					end)

					if not __ok or not __svc then
						return false, 'missing ' .. __name
					end
				end

				return true, 'all services ok'
			end,
		},
		{
			name = 'LocalPlayer',
			run = function()
				local __lp = Players.LocalPlayer

				if not __lp then
					Players:GetPropertyChangedSignal('LocalPlayer'):Wait()
					__lp = Players.LocalPlayer
				end

				if not __lp then
					return false, 'LocalPlayer nil'
				end

				return true, __lp.Name .. ' (' .. tostring(__lp.UserId) .. ')'
			end,
		},
		{
			name = 'PlayerGui',
			run = function()
				local __lp = Players.LocalPlayer

				if not __lp then
					return false, 'no player'
				end

				local __pg = __lp:FindFirstChildOfClass('PlayerGui') or __lp:WaitForChild('PlayerGui', 8)

				if not __pg then
					return false, 'PlayerGui timeout'
				end

				return true, 'PlayerGui ready'
			end,
		},
		{
			name = 'Character / workspace',
			run = function()
				local __lp = Players.LocalPlayer

				if not __lp then
					return false, 'no player'
				end

				local __char = __lp.Character or __lp.CharacterAdded:Wait()

				if not __char then
					return false, 'no character'
				end

				local __root = __char:FindFirstChild('HumanoidRootPart') or __char:WaitForChild('HumanoidRootPart', 5)
				local __hum = __char:FindFirstChildOfClass('Humanoid')

				if not __root or not __hum then
					return false, 'rig incomplete'
				end

				if not workspace or not workspace.CurrentCamera then
					return false, 'camera missing'
				end

				return true, 'character + camera ok'
			end,
		},
		{
			name = 'Executor env',
			run = function()
				local __bits = {}

				if typeof(cloneref) == 'function' then
					table.insert(__bits, 'cloneref')
				end

				if typeof(getgenv) == 'function' then
					table.insert(__bits, 'getgenv')
				end

				if typeof(gethui) == 'function' then
					table.insert(__bits, 'gethui')
				end

				if typeof(hookmetamethod) == 'function' then
					table.insert(__bits, 'hmm')
				end

				if #__bits == 0 then
					return true, 'vanilla / limited env'
				end

				return true, table.concat(__bits, ', ')
			end,
		},
		{
			name = 'Heartbeat / render',
			run = function()
				local __ok = false
				local __conn

				__conn = RunService.Heartbeat:Connect(function()
					__ok = true
				end)

				local __t0 = os.clock()

				while not __ok and os.clock() - __t0 < 2 do
					task.wait()
				end

				if __conn then
					__conn:Disconnect()
				end

				if not __ok then
					return false, 'no heartbeat'
				end

				return true, 'render loop alive'
			end,
		},
		{
			name = 'CoreGui access',
			run = function()
				local __ok, __cg = pcall(function()
					return game:GetService('CoreGui')
				end)

				if not __ok or not __cg then
					return false, 'CoreGui blocked'
				end

				local __probe = Instance.new('Folder')
				__probe.Name = '__euphoria_probe'

				local __parent_ok, __err = pcall(function()
					__probe.Parent = __cg
				end)

				if __probe.Parent then
					__probe:Destroy()
				else
					__probe:Destroy()
				end

				if not __parent_ok then
					return false, 'cant parent CoreGui: ' .. tostring(__err)
				end

				return true, 'CoreGui writable'
			end,
		},
	}
end

function Console:run() : boolean
	if self.__running then
		return false
	end

	self.__running = true

	local __brand = self.__brand

	self:blank()
	self:header({
		'Welcome to ' .. __brand .. '!',
		'Main Developers: riq & enrithy',
		'Contributors: community',
		'',
		'Join Us: discord.gg/euphoria',
	})

	self:spinner('Booting', 0.55)
	self:status('Checking Script STATUS:', 'WORKING')
	self:blank()

	local __steps = self.__steps

	if #__steps == 0 then
		__steps = __default_steps()
	end

	local __total = #__steps
	local __passed = 0
	local __failed = 0

	for __i, __step in ipairs(__steps) do
		self:status('Running → ' .. __step.name, '...')

		local __success = false
		local __msg = 'ok'

		local __call_ok, __a, __b = pcall(function()
			return __step.run()
		end)

		if not __call_ok then
			__success = false
			__msg = tostring(__a)
		else
			__success = __a == true
			__msg = __b or (__success and 'ok' or 'failed')
		end

		if __success then
			__passed += 1
			self:status(__step.name, 'OK')
			self:timestamp_log(tostring(__msg))
		else
			__failed += 1
			self:status(__step.name, 'FAIL')
			self:warn_line(self.__prefix .. ': ' .. tostring(__msg))
		end

		self:progress((__i / __total) * 100)
		task.wait(self.__delay * 1.4)
	end

	self:blank()

	if __failed > 0 then
		self:status('Load finished with issues', 'WARN')
		self:warn_line(self.__prefix .. ': ' .. tostring(__passed) .. '/' .. tostring(__total) .. ' steps ok, ' .. tostring(__failed) .. ' failed')
	else
		self:status('Load finished', 'READY')
		self:timestamp_log('all systems green')
	end

	self:progress(100)
	self:blank()
	self:timestamp_log(__brand .. ' console online — open F9 to see this')

	self.__running = false

	return __failed == 0
end

local function clamp(n: any, a: number, b: number): number
	n = tonumber(n) or a

	return math.max(a, math.min(b, n))
end

local function notify(self: EnrithyReach, msg: string, kind: string?): ()
	local lib = self.__library

	if type(lib) == 'table' and type(lib.notify) == 'function' then
		pcall(function()
			lib:notify({
				title = 'enrithy',
				content = msg,
				duration = 2.3,
				notify_type = kind or 'normal',
			})
		end)
	end
end

local function isWorkspace(inst: any): boolean
	if inst == Workspace or inst == workspace then
		return true
	end

	if typeof(inst) == 'Instance' then
		return inst.ClassName == 'Workspace'
	end

	return false
end

local function killPrevious()
	local prev = rawget(_G, KEY)

	if type(prev) == 'table' then
		prev.dead = true

		pcall(function()
			if prev.charConn then
				prev.charConn:Disconnect()
			end
		end)

		pcall(function()
			if prev.hbConn then
				prev.hbConn:Disconnect()
			end
		end)

		pcall(function()
			if prev.vizHb then
				prev.vizHb:Disconnect()
			end
		end)

		pcall(function()
			if prev.viz then
				prev.viz:Destroy()
			end
		end)

		if type(prev.restoreCG) == 'function' then
			pcall(prev.restoreCG)
		end
	end

	pcall(function()
		for _, c in ipairs(Workspace.Terrain:GetChildren()) do
			if c.Name == VIZ_NAME or c.Name == 'TouchlineReachViz' then
				c:Destroy()
			end
		end
	end)

	local old = rawget(_G, '_TouchlineReach')

	if type(old) == 'table' then
		old.dead = true

		if type(old.restoreCG) == 'function' then
			pcall(old.restoreCG)
		end
	end
end

local function getChar(self: EnrithyReach): Model?
	local c = self.char

	if c and c.Parent then
		return c
	end

	local folder = Workspace:FindFirstChild('Characters')

	if folder then
		c = folder:FindFirstChild(LocalPlayer.Name) :: Model?
	end

	if not c then
		c = LocalPlayer.Character
	end

	self.char = c
	self.hrp = nil
	self.bodyParts = nil

	return c
end

local function getHrp(self: EnrithyReach): BasePart?
	local hrp = self.hrp

	if hrp and hrp.Parent then
		return hrp
	end

	local char = getChar(self)
	hrp = char and (char:FindFirstChild('HumanoidRootPart') :: BasePart?) or nil
	self.hrp = hrp

	return hrp
end

local function getBodyParts(self: EnrithyReach): { BasePart }
	local cached = self.bodyParts

	if cached then
		local ok = true

		for i = 1, #cached do
			if not cached[i] or not cached[i].Parent then
				ok = false
				break
			end
		end

		if ok then
			return cached
		end
	end

	local char = getChar(self)
	local parts = {}

	if not char then
		return parts
	end

	for i = 1, #BODY_NAMES do
		local p = char:FindFirstChild(BODY_NAMES[i])

		if p and p:IsA('BasePart') then
			parts[#parts + 1] = p
		end
	end

	-- any other BaseParts on character (custom rigs)
	if #parts < 3 then
		for _, d in ipairs(char:GetDescendants()) do
			if d:IsA('BasePart') then
				local found = false

				for i = 1, #parts do
					if parts[i] == d then
						found = true
						break
					end
				end

				if not found then
					parts[#parts + 1] = d
				end
			end
		end
	end

	self.bodyParts = parts

	return parts
end

local function footballs(self: EnrithyReach): Folder?
	local f = self.folder

	if f and f.Parent then
		return f
	end

	f = Workspace:FindFirstChild('Footballs') :: Folder?

	if not f then
		-- some places nest it
		for _, d in ipairs(Workspace:GetChildren()) do
			if d.Name == 'Footballs' and d:IsA('Folder') then
				f = d
				break
			end
		end
	end

	self.folder = f

	return f
end

local function isPlayBall(p: any): boolean
	if typeof(p) ~= 'Instance' or not p:IsA('BasePart') then
		return false
	end

	if p.Anchored then
		return false
	end

	if not p.Parent then
		return false
	end

	-- strict original
	if p.Name == 'Ball' and p.Parent.Name == 'Footballs' then
		return true
	end

	-- loose fallback names
	if p.Parent.Name == 'Footballs' then
		local n = string.lower(p.Name)

		if n == 'ball' or n == 'football' or n == 'soccerball' then
			return true
		end
	end

	return false
end

local function distToBody(ball: BasePart, parts: { BasePart }): number
	local best = math.huge
	local pos = ball.Position

	for i = 1, #parts do
		local d = (pos - parts[i].Position).Magnitude

		if d < best then
			best = d
		end
	end

	return best
end

local function ballInReach(self: EnrithyReach, ball: BasePart, studs: number): boolean
	local hrp = getHrp(self)

	if not hrp then
		return false
	end

	local parts = getBodyParts(self)

	if #parts == 0 then
		parts = { hrp }
	end

	if distToBody(ball, parts) <= studs then
		return true
	end

	local localPos = hrp.CFrame:PointToObjectSpace(ball.Position)

	if localPos.Y > 0.25 then
		local char = getChar(self)
		local head = char and char:FindFirstChild('Head')

		if head and head:IsA('BasePart') then
			if (ball.Position - head.Position).Magnitude <= studs then
				return true
			end
		end

		local xz = math.sqrt(localPos.X * localPos.X + localPos.Z * localPos.Z)

		if xz <= studs and localPos.Y <= studs + 2 then
			return true
		end
	end

	return false
end

local function nearestBall(self: EnrithyReach, studs: number): BasePart?
	local folder = footballs(self)
	local hrp = getHrp(self)

	if not hrp then
		return nil
	end

	local parts = getBodyParts(self)

	if #parts == 0 then
		parts = { hrp }
	end

	local best: BasePart? = nil
	local bestD = math.huge

	local function consider(child: BasePart)
		if not ballInReach(self, child, studs) then
			return
		end

		local d = distToBody(child, parts)
		local localY = hrp.CFrame:PointToObjectSpace(child.Position).Y

		if localY > 0.25 then
			local char = getChar(self)
			local head = char and char:FindFirstChild('Head')

			if head and head:IsA('BasePart') then
				d = math.min(d, (child.Position - head.Position).Magnitude)
			end
		end

		if d < bestD then
			bestD = d
			best = child
		end
	end

	if folder then
		for _, child in ipairs(folder:GetChildren()) do
			if isPlayBall(child) then
				consider(child :: BasePart)
			end
		end
	else
		-- last resort scan
		for _, inst in ipairs(Workspace:GetDescendants()) do
			if inst:IsA('BasePart') and inst.Name == 'Ball' and not inst.Anchored then
				consider(inst)
			end
		end
	end

	return best
end

local function isTouchHitbox(size: any): boolean
	if typeof(size) ~= 'Vector3' then
		-- if size missing, still allow inject (some calls differ)
		return true
	end

	local s = size :: Vector3

	-- original band
	if s.X >= 1.5 and s.X <= 8 and s.Y >= 2.5 and s.Y <= 10 and s.Z >= 1 and s.Z <= 10 then
		return true
	end

	-- slightly looser (game updates)
	if s.X >= 1 and s.X <= 12 and s.Y >= 1.5 and s.Y <= 14 and s.Z >= 0.8 and s.Z <= 12 then
		return true
	end

	return false
end

local function inject(self: EnrithyReach, results: any, studs: number): any
	local ball = nearestBall(self, studs)

	if not ball then
		return results
	end

	if type(results) ~= 'table' then
		results = { ball }
		self.__hits += 1

		return results
	end

	for i = 1, #results do
		if results[i] == ball then
			return results
		end
	end

	results[#results + 1] = ball
	self.__hits += 1

	return results
end

local function saveAndSetCG(self: EnrithyReach, char: Model, group: string)
	for _, p in ipairs(char:GetDescendants()) do
		if p:IsA('BasePart') then
			if self.savedCG[p] == nil then
				pcall(function()
					self.savedCG[p] = p.CollisionGroup
				end)
			end

			pcall(function()
				p.CollisionGroup = group
			end)
		end
	end
end

local function restoreCG(self: EnrithyReach)
	for part, group in pairs(self.savedCG) do
		if part and part.Parent then
			pcall(function()
				part.CollisionGroup = group
			end)
		end
	end

	table.clear(self.savedCG)
end

local function applyBallCol(self: EnrithyReach)
	local char = getChar(self)

	if not char then
		return
	end

	if self.ballCol then
		if next(self.savedCG) then
			restoreCG(self)
		else
			saveAndSetCG(self, char, 'Player')
			restoreCG(self)
		end
	else
		saveAndSetCG(self, char, 'SelfPhase')
	end
end

local function destroyViz(self: EnrithyReach)
	if self.vizHb then
		pcall(function()
			self.vizHb:Disconnect()
		end)
		self.vizHb = nil
	end

	if self.viz then
		pcall(function()
			self.viz:Destroy()
		end)
		self.viz = nil
	end
end

local function ensureViz(self: EnrithyReach)
	destroyViz(self)

	if not self.showViz or self.dead then
		return
	end

	local part = Instance.new('Part')
	part.Name = VIZ_NAME
	part.Shape = Enum.PartType.Ball
	part.Anchored = true
	part.CanCollide = false
	part.CanTouch = false
	part.CanQuery = false
	part.CastShadow = false
	part.Material = Enum.Material.ForceField
	part.Color = Color3.fromRGB(255, 255, 255)
	part.Transparency = 0.85
	part.Parent = Workspace.Terrain
	self.viz = part

	self.vizHb = RunService.RenderStepped:Connect(function()
		if self.dead or not self.showViz or not self.viz then
			return
		end

		local hrp = getHrp(self)

		if not hrp then
			return
		end

		local d = clamp(self.studs, 1, MAX_STUDS) * 2
		self.viz.Size = Vector3.new(d, d, d)
		self.viz.CFrame = CFrame.new(hrp.Position)
		self.viz.Transparency = self.enabled and 0.82 or 0.95
	end)
end

-- ─── PRIMARY: bounds inject ───────────────────────────────────────

local function process(selfWs: any, cf: any, size: any, params: any, callOld: any): any
	local results = callOld(selfWs, cf, size, params)
	local st = rawget(_G, KEY)

	if type(st) ~= 'table' or st.dead or not st.enabled or not st.useHook then
		return results
	end

	if not isTouchHitbox(size) then
		return results
	end

	-- character present somehow
	if not getHrp(st) then
		return results
	end

	return inject(st, results, clamp(st.studs, 1, MAX_STUDS))
end

local function installHooks(self: EnrithyReach)
	rawset(_G, '_EnrithyReachProcess', process)
	rawset(_G, '_TouchlineReachProcess', process)

	local hooked = false

	-- 1) namecall hook (main)
	if not rawget(_G, '_EnrithyReachNC') and typeof(hookmetamethod) == 'function' and typeof(getnamecallmethod) == 'function' then
		local old
		local calling = false

		local function handler(selfWs: any, ...: any): any
			if calling then
				return old(selfWs, ...)
			end

			local method = getnamecallmethod()

			if (method == 'GetPartBoundsInBox' or method == 'getpartsboundinbox') and isWorkspace(selfWs) then
				local cf, size, params = ...
				local fn = rawget(_G, '_EnrithyReachProcess')

				if type(fn) == 'function' then
					calling = true

					local ok, a, b, c = pcall(function()
						return fn(selfWs, cf, size, params, function(s: any, cframe: any, sz: any, pr: any)
							return old(s, cframe, sz, pr)
						end)
					end)

					calling = false

					if ok then
						return a, b, c
					end
				end
			end

			return old(selfWs, ...)
		end

		if typeof(newcclosure) == 'function' then
			-- some executors break with newcclosure on namecall — try raw first
			local okHook, errHook = pcall(function()
				old = hookmetamethod(game, '__namecall', handler)
			end)

			if not okHook then
				pcall(function()
					old = hookmetamethod(game, '__namecall', newcclosure(handler))
				end)
			end
		else
			pcall(function()
				old = hookmetamethod(game, '__namecall', handler)
			end)
		end

		if old then
			rawset(_G, '_EnrithyReachNC', true)
			hooked = true
		else
			warn('[enrithy] namecall hook failed')
		end
	elseif rawget(_G, '_EnrithyReachNC') then
		hooked = true
	end

	-- 2) direct method hook backup
	if not rawget(_G, '_EnrithyReachFN') and typeof(hookfunction) == 'function' then
		local okFn = pcall(function()
			local real = Workspace.GetPartBoundsInBox

			if typeof(real) ~= 'function' then
				return
			end

			local oldFn
			oldFn = hookfunction(real, function(selfWs: any, cf: any, size: any, params: any)
				local fn = rawget(_G, '_EnrithyReachProcess')

				if type(fn) == 'function' then
					local ok, a = pcall(function()
						return fn(selfWs, cf, size, params, function(s: any, cframe: any, sz: any, pr: any)
							return oldFn(s, cframe, sz, pr)
						end)
					end)

					if ok then
						return a
					end
				end

				return oldFn(selfWs, cf, size, params)
			end)

			rawset(_G, '_EnrithyReachFN', true)
			hooked = true
		end)

		if not okFn then
			-- ignore
		end
	elseif rawget(_G, '_EnrithyReachFN') then
		hooked = true
	end

	self.__hookOk = hooked

	if not hooked then
		warn('[enrithy] NO HOOK — using firetouch backup only')
	end
end

-- ─── BACKUP: firetouchinterest (TPS-style) ────────────────────────

local function fireTouch(part: BasePart, ball: BasePart)
	if typeof(firetouchinterest) ~= 'function' then
		return
	end

	pcall(function()
		firetouchinterest(part, ball, 0)
		firetouchinterest(part, ball, 1)
	end)
end

local function touchStep(self: EnrithyReach)
	if self.dead or not self.enabled or not self.useTouch then
		return
	end

	if typeof(firetouchinterest) ~= 'function' then
		return
	end

	local studs = clamp(self.studs, 1, MAX_STUDS)
	local ball = nearestBall(self, studs)

	if not ball then
		return
	end

	local parts = getBodyParts(self)
	local hrp = getHrp(self)

	if hrp and #parts == 0 then
		parts = { hrp }
	end

	local now = os.clock()

	for i = 1, #parts do
		local part = parts[i]

		if not part.Parent then
			continue
		end

		if (part.Position - ball.Position).Magnitude > studs + 1.5 then
			continue
		end

		local key = tostring(part) .. ':' .. tostring(ball)
		local last = self.lastTouch[key] or 0

		if now - last < 0.03 then
			continue
		end

		self.lastTouch[key] = now
		fireTouch(part, ball)
		self.__hits += 1
	end
end

	killPrevious()

	self.dead = false
	self.enabled = true
	self.studs = DEFAULT_STUDS
	self.ballCol = true
	self.showViz = false
	self.useHook = true
	self.useTouch = true
	self.folder = nil
	self.hrp = nil
	self.char = nil
	self.bodyParts = nil
	self.charConn = nil
	self.hbConn = nil
	self.viz = nil
	self.vizHb = nil
	self.savedCG = {}
	self.lastTouch = {}
	self.__library = nil
	self.__started = false
	self.__hookOk = false
	self.__hits = 0

	self.restoreCG = function()
		restoreCG(self)
	end

	rawset(_G, KEY, self)
	rawset(_G, 'EnrithyReachApi', self)

	return self
end

function EnrithyReach:set_enabled(state: boolean)
	self.enabled = state and true or false
end

function EnrithyReach:set_studs(n: number)
	self.studs = clamp(n, 1, MAX_STUDS)
end

	self.ballCol = on and true or false
	applyBallCol(self)
end

function EnrithyReach:set_viz(on: boolean)
	self.showViz = on and true or false

	if self.showViz then
		ensureViz(self)
	else
		destroyViz(self)
	end

function EnrithyReach:status()
	local c = getChar(self)
	local f = footballs(self)
	local balls = 0

	if f then
		for _, ch in ipairs(f:GetChildren()) do
			if isPlayBall(ch) then
				balls += 1
			end
		end
	end

	return {
		enabled = self.enabled,
		studs = self.studs,
		ballCol = self.ballCol,
		showViz = self.showViz,
		char = c and c.Name or nil,
		hooked = self.__hookOk,
		footballs = f ~= nil,
		balls = balls,
		hits = self.__hits,
		hasFTI = typeof(firetouchinterest) == 'function',
	}
end

function EnrithyReach:start()
	if self.dead then
		return
	end

	rawset(_G, KEY, self)
	rawset(_G, '_EnrithyReachProcess', process)
	rawset(_G, '_TouchlineReachProcess', process)

	if not self.__started then
		self.__started = true
		installHooks(self)

		self.charConn = LocalPlayer.CharacterAdded:Connect(function()
			self.char = nil
			self.hrp = nil
			self.bodyParts = nil
			table.clear(self.savedCG)
			table.clear(self.lastTouch)

			task.defer(function()
				if self.dead then
					return
				end

				getChar(self)

				if not self.ballCol then
					applyBallCol(self)
				end

				if self.showViz then
					ensureViz(self)
				end
			end)
		end)

		self.hbConn = RunService.Heartbeat:Connect(function()
			if self.dead then
				return
			end

			local ok, err = pcall(touchStep, self)

			if not ok then
				warn('[enrithy touch] ' .. tostring(err))
			end
		end)
	else
		installHooks(self)
	end

	getChar(self)
	self:set_ball_col(self.ballCol)

	local s = self:status()
	print(string.format(
		'[enrithy] ready | studs=%s hook=%s fti=%s footballs=%s balls=%s',
		tostring(s.studs),
		tostring(s.hooked),
		tostring(s.hasFTI),
		tostring(s.footballs),
		tostring(s.balls)
	))
end

function EnrithyReach:cleanup()
	self.dead = true
	self.enabled = false
	self.__started = false

	if self.charConn then
		pcall(function()
			self.charConn:Disconnect()
		end)
		self.charConn = nil
	end

	if self.hbConn then
		pcall(function()
			self.hbConn:Disconnect()
		end)
		self.hbConn = nil
	end

	destroyViz(self)
	restoreCG(self)

	rawset(_G, '_EnrithyReachProcess', nil)
	rawset(_G, '_TouchlineReachProcess', nil)
	rawset(_G, 'EnrithyReachApi', nil)

	if rawget(_G, KEY) == self then
		rawset(_G, KEY, nil)
	end
end

function EnrithyReach:stop()
	self:cleanup()
end

	self.__library = library

		end,

	tab:create_module({
		title = 'Ball collision',
		default = true,
		callback = function(state)
			self_:set_ball_col(state)
		end,

	tab:create_module({
		title = 'Show radius',
		default = false,
		callback = function(state)
			self_:set_viz(state)

	})

	tab:create_button({
		title = 'Status',
		callback = function()
			local s = self_:status()

				'on=%s studs=%s col=%s char=%s',

				tostring(s.ballCol),
				tostring(s.char or '?')
			)
			notify(self_, line)
		end,
	})
end

	self:cleanup()
	self.__library = nil
end

local function __clamp_ms(__v: number): number
	return math.clamp(math.floor(__v + 0.5), __MIN_MS, __MAX_MS)
end

local function __color_for(__ms: number): Color3
	if __ms <= 90 then
		return Color3.fromRGB(0, 255, 127)
	elseif __ms <= 150 then
		return Color3.fromRGB(255, 255, 0)
	elseif __ms <= 200 then
		return Color3.fromRGB(255, 170, 0)
	end

	return Color3.fromRGB(255, 0, 0)
end

-- TPS-style backpack value (if exists)
local function __write_backpack(__ms: number): ()
	local __pack = LocalPlayer:FindFirstChild('Backpack')
	local __ping = __pack and __pack:FindFirstChild('Ping')

	if __ping and __ping:IsA('IntValue') and __ping.Value ~= __ms then
		pcall(function()
			__ping.Value = __ms
		end)
	end
end

-- Touchline in-game label: PlayerGui.Main.Game.Statistics.Ping
local function __find_game_ping(): TextLabel?
	local __pg = LocalPlayer:FindFirstChildOfClass('PlayerGui')
	if not __pg then
		return nil
	end

	local __main = __pg:FindFirstChild('Main')
	local __game = __main and __main:FindFirstChild('Game')
	local __stats = __game and __game:FindFirstChild('Statistics')
	local __ping = __stats and __stats:FindFirstChild('Ping')

	if __ping and __ping:IsA('TextLabel') then
		return __ping
	end

	-- soft search
	for _, __d in ipairs(__pg:GetDescendants()) do
		if __d:IsA('TextLabel') and __d.Name == 'Ping' then
			local __t = string.lower(__d.Text)
			if __t:find('ping') or __t == '' or __t:find('ms') or __t:find('nil') then
				return __d
			end
		end
	end

	return nil
end

-- Roblox PerformanceStats NetworkPing panel (value + bar graph)
local function __find_network_ping_panel(): (TextLabel?, Instance?, Frame?)
	local __ok, __ps = pcall(function()
		local __rg = CoreGui:FindFirstChild('RobloxGui')
		return __rg and __rg:FindFirstChild('PerformanceStats')
	end)

	if not __ok or not __ps then
		return nil, nil, nil
	end

	for _, __d in ipairs(__ps:GetDescendants()) do
		if __d:IsA('TextLabel') and __d.Name == 'TitleLabel' and __d.Text == 'NetworkPing' then
			local __panel = __d.Parent
			local __val = __panel and __panel:FindFirstChild('ValueLabel')
			local __btn = __panel and __panel.Parent -- PS_Button
			local __bar_frame: Frame? = nil

			if __btn then
				local __graph = __btn:FindFirstChild('PS_AnnotatedGraph')
				local __bf = __graph and __graph:FindFirstChild('PS_BarFrame')

				if __bf and __bf:IsA('Frame') then
					__bar_frame = __bf
				end
			end

			if __val and __val:IsA('TextLabel') then
				return __val, __btn, __bar_frame
			end
		end
	end

	return nil, nil, nil
end

-- Official Roblox PerformanceStats bar colors (CoreScripts StatsUtils.lua)
-- GraphBarGreenColor  = Color3.new(126/255, 211/255, 33/255)
-- GraphBarYellowColor = Color3.new(209/255, 211/255, 33/255)
-- GraphBarRedColor    = Color3.new(211/255, 88/255, 33/255)
-- GetColorForValue(value, target): <0.666*target green, <1.333 yellow, else red
-- Ping target used by widget is typically ~100 ms
local __BAR_OK = Color3.fromRGB(126, 211, 33)
local __BAR_MID = Color3.fromRGB(209, 211, 33)
local __BAR_BAD = Color3.fromRGB(211, 88, 33)
local __PING_TARGET = 100

local function __bar_color(__ms: number): Color3
	if __ms < 0.666 * __PING_TARGET then
		return __BAR_OK
	elseif __ms < 1.333 * __PING_TARGET then
		return __BAR_MID
	end

	return __BAR_BAD
end

local function __paint_bars(__bar_frame: Frame?, __ms: number): ()
	if not __bar_frame or not __bar_frame.Parent then
		return
	end

	local __frac = 0.06

	-- stock NetworkPing looks like a thin teal strip
	if __ms <= 30 then
		__frac = 0.08
	elseif __ms <= 60 then
		__frac = 0.12
	elseif __ms <= 90 then
		__frac = 0.18
	else
		__frac = math.clamp(__ms / 350, 0.22, 0.6)
	end

	local __col = __bar_color(__ms)

	for _, __b in ipairs(__bar_frame:GetChildren()) do
		if __b:IsA('Frame') and string.sub(__b.Name, 1, 4) == 'Bar_' then
			pcall(function()
				__b.BackgroundColor3 = __col
				__b.BackgroundTransparency = 0
				local __x = __b.Size.X
				__b.Size = UDim2.new(__x.Scale, __x.Offset, __frac, 0)
				__b.AnchorPoint = Vector2.new(__b.AnchorPoint.X, 1)
				local __px = __b.Position.X
				__b.Position = UDim2.new(__px.Scale, __px.Offset, 1, 0)
			end)
		end
	end
end

local function __ensure_hook(__self: PingChanger): ()
	if __self.__hooked then
		return
	end

	if typeof(hookfunction) ~= 'function' then
		return
	end

	local __ok, __err = pcall(function()
		local __method = LocalPlayer.GetNetworkPing

		if typeof(__method) ~= 'function' then
			return
		end

		__self.__old_gnp = hookfunction(__method, function(__player, ...)
			if __self.__enabled and __player == LocalPlayer then
				return __self.__ms / 1000
			end

			if __self.__old_gnp then
				return __self.__old_gnp(__player, ...)
			end

			return 0.05
		end)

		__self.__hooked = true
	end)

	if not __ok then
		warn('[enrithy/euphoria ping] hook ' .. tostring(__err))
	end
end

local function __paint(__self: PingChanger): ()
	local __ms = __self.__ms
	local __now = os.clock()

	-- rescan every 0.75s (PerformanceStats rebuilds bars often)
	if __now - __self.__scan_t > 0.75
		or not __self.__net_value
		or not __self.__net_value.Parent
		or not __self.__bar_frame
		or not __self.__bar_frame.Parent
	then
		__self.__scan_t = __now

		local __val, __btn, __bars = __find_network_ping_panel()
		__self.__net_value = __val
		__self.__net_btn = __btn :: any
		__self.__bar_frame = __bars
		__self.__game_label = __find_game_ping()
	end

	if not __self.__game_label or not __self.__game_label.Parent then
		__self.__game_label = __find_game_ping()
	end

	local __netText = tostring(__ms) .. ' ms'
	local __gameText = 'Ping: ' .. tostring(__ms) .. ' ms'
	local __col = __color_for(__ms)

	if __self.__net_value then
		pcall(function()
			-- text only — keep Roblox default color (don't paint value green/red)
			__self.__net_value.Text = __netText
		end)
	end

	-- flatten / recolor the graph bars only
	__paint_bars(__self.__bar_frame, __ms)

	if __self.__game_label then
		pcall(function()
			__self.__game_label.Text = __gameText
			__self.__game_label.TextColor3 = __col
		end)
	end

	__write_backpack(__ms)
end

function PingChanger:init(): PingChanger
	self.__enabled = false
	self.__ms = __DEFAULT_MS
	self.__last = 0
	self.__library = nil
	self.__net_value = nil
	self.__game_label = nil
	self.__net_btn = nil
	self.__bar_frame = nil
	self.__scan_t = 0

	return self
end

function PingChanger:set_ms(__value: number): ()
	self.__ms = __clamp_ms(__value)

	if self.__enabled then
		__paint(self)
	end
end

function PingChanger:set_enabled(__state: boolean): ()
	self.__enabled = __state and true or false

	if self.__enabled then
		__ensure_hook(self)
		self.__scan_t = 0
		__paint(self)
	end
end

function PingChanger:start(): ()
	if self.__running then
		return
	end

	self.__running = true
	self.__last = 0
	__ensure_hook(self)

	-- RenderStepped so we win against Roblox PerformanceStats refresh
	self.__conn = RunService.RenderStepped:Connect(function()
		if not self.__enabled then
			return
		end

		local __ok, __err = pcall(__paint, self)

		if not __ok then
			warn('[ping] ' .. tostring(__err))
		end
	end)
end

function PingChanger:stop(): ()
	self.__enabled = false

	if self.__conn then
		self.__conn:Disconnect()
		self.__conn = nil
	end

	self.__running = false
end

function PingChanger:attach_ui(__library: any): ()
	if type(__library) ~= 'table' or type(__library.create_tab) ~= 'function' then
		return
	end

	self.__library = __library

	local __self = self
	local __tab = __library:create_tab('Ping', 'rbxassetid://10734950309')

	__tab:create_divider('Ping display')

	__tab:create_module({
		title = 'Ping Changer',
		default = false,
		callback = function(__state)
			__self:set_enabled(__state)
			__notify(__self, if __state then 'on · ' .. tostring(__self.__ms) .. ' ms' else 'off')
		end,
	})

	__tab:create_slider({
		title = 'Ping (ms)',
		minimum = __MIN_MS,
		maximum = __MAX_MS,
		default = __DEFAULT_MS,
		rounding = 1,
		callback = function(__v)
			__self:set_ms(__v)
		end,
	})

	__tab:create_divider('Presets')

	local function __preset(__ms: number)
		__self:set_ms(__ms)

		if not __self.__enabled then
			__self:set_enabled(true)
		end

		__notify(__self, tostring(__ms) .. ' ms')
	end

	__tab:create_button({
		title = 'Preset · 4 ms',
		callback = function()
			__preset(4)
		end,
	})

	__tab:create_button({
		title = 'Preset · 5 ms',
		callback = function()
			__preset(5)
		end,
	})

	__tab:create_button({
		title = 'Preset · 10 ms',
		callback = function()
			__preset(10)
		end,
	})

	__tab:create_button({
		title = 'Preset · 20 ms',
		callback = function()
			__preset(20)
		end,
	})

	__tab:create_button({
		title = 'Status',
		callback = function()
			local __net, _, __bars = __find_network_ping_panel()
			local __game = __find_game_ping()
			__notify(
				__self,
				string.format(
					'on=%s ms=%s netUI=%s bars=%s gameUI=%s',
					tostring(__self.__enabled),
					tostring(__self.__ms),
					tostring(__net ~= nil),
					tostring(__bars ~= nil),
					tostring(__game ~= nil)
				)
			)
		end,
	})
end

function PingChanger:destroy(): ()
	self:stop()
	self.__library = nil
end

local function __wait_character(__player: Player, __timeout: number?): Model?
	local __t = __timeout or 5
	local __char = __player.Character

	if __char and __humanoid(__char) then
		return __char
	end

	local __deadline = os.clock() + __t

	while os.clock() < __deadline do
		__char = __player.Character

		if __char and __humanoid(__char) then
			return __char
		end

		task.wait(0.05)
	end

	return __player.Character
end

local function __asset_uri(__id: number): string

local function __parse_query(__query: string | number | Player): (string?, number?, Player?)
	if typeof(__query) == 'Instance' and (__query :: Instance):IsA('Player') then
		local __p = __query :: Player

		return __p.Name, __p.UserId, __p
	end

	if type(__query) == 'number' then
		local __id = math.floor(__query :: number)

		if __id <= 0 then
			return nil, nil, nil
		end

		for _, __p in ipairs(Players:GetPlayers()) do
			if __p.UserId == __id then
				return __p.Name, __id, __p
			end
		end

		return tostring(__id), __id, nil
	end

	local __raw = __trim(__query)

	if __raw == '' then
		return nil, nil, nil
	end

	if string.sub(__raw, 1, 1) == '@' then
		__raw = __trim(string.sub(__raw, 2))
	end

	local __url_id = string.match(__raw, 'roblox%.com/users/(%d+)') or string.match(__raw, 'users/(%d+)')

	if __url_id then
		return __parse_query(tonumber(__url_id) :: number)
	end

	local __as_num = tonumber(__raw)

	if __as_num and __as_num == math.floor(__as_num) and __as_num > 0 and not string.find(__raw, '%a') then
		return __parse_query(__as_num)
	end

	local __lower = string.lower(__raw)

	for _, __p in ipairs(Players:GetPlayers()) do
		if string.lower(__p.Name) == __lower or string.lower(__p.DisplayName) == __lower then
			return __p.Name, __p.UserId, __p
		end
	end

	for _, __p in ipairs(Players:GetPlayers()) do
		if string.find(string.lower(__p.Name), __lower, 1, true)
			or string.find(string.lower(__p.DisplayName), __lower, 1, true)
		then
			return __p.Name, __p.UserId, __p
		end
	end

	local __ok, __user_id = pcall(function()
		return Players:GetUserIdFromNameAsync(__raw)
	end)

	if __ok and type(__user_id) == 'number' and __user_id > 0 then
		for _, __p in ipairs(Players:GetPlayers()) do
			if __p.UserId == __user_id then
				return __p.Name, __user_id, __p
			end
		end

		local __name_ok, __name = pcall(function()
			return Players:GetNameFromUserIdAsync(__user_id)
		end)

		return (if __name_ok and type(__name) == 'string' then __name else __raw), __user_id, nil
	end

	return nil, nil, nil
end

local function __split_csv_ids(__csv: string): { number }
	local __ids = {}
	local __seen: { [number]: boolean } = {}

	for __piece in string.gmatch(tostring(__csv or ''), '[^%,]+') do
		local __n = tonumber(__trim(__piece))

		if __n and __n > 0 then
			local __id = math.floor(__n)

			if not __seen[__id] then
				__seen[__id] = true
				table.insert(__ids, __id)
			end
		end
	end

	return __ids
end

local function __collect_accessory_ids(__desc: HumanoidDescription): { number }
	local __ids = {}
	local __seen: { [number]: boolean } = {}

	local function __push(__id: number)
		if __id > 0 and not __seen[__id] then
			__seen[__id] = true
			table.insert(__ids, __id)
		end
	end

	-- modern API
	pcall(function()
		local __list = (__desc :: any):GetAccessories(true)

		if type(__list) == 'table' then
			for _, __item in ipairs(__list) do
				if type(__item) == 'table' then
					local __aid = tonumber((__item :: any).AssetId or (__item :: any).assetId)

					if __aid then
						__push(math.floor(__aid))
					end
				elseif typeof(__item) == 'Instance' then
					local __aid = tonumber((__item :: any).AssetId)

					if __aid then
						__push(math.floor(__aid))
					end
				end
			end
		end
	end)

	-- also layered / false flag
	pcall(function()
		local __list = (__desc :: any):GetAccessories(false)

		if type(__list) == 'table' then
			for _, __item in ipairs(__list) do
				if type(__item) == 'table' then
					local __aid = tonumber((__item :: any).AssetId or (__item :: any).assetId)

					if __aid then
						__push(math.floor(__aid))
					end
				elseif typeof(__item) == 'Instance' then
					local __aid = tonumber((__item :: any).AssetId)

					if __aid then
						__push(math.floor(__aid))
					end
				end
			end
		end
	end)

	local __fields = {
		__desc.HatAccessory,
		__desc.HairAccessory,
		__desc.FaceAccessory,
		__desc.NeckAccessory,
		__desc.ShouldersAccessory,
		__desc.FrontAccessory,
		__desc.BackAccessory,
		__desc.WaistAccessory,
	}

	for _, __field in ipairs(__fields) do
		for _, __id in ipairs(__split_csv_ids(__field)) do
			__push(__id)
		end
	end

	return __ids
end

local function __prepare_handle(__handle: BasePart): ()
	__handle.Anchored = false
	__handle.CanCollide = false
	__handle.CanTouch = false
	__handle.CanQuery = false
	__handle.Massless = true
	pcall(function()
		__handle.CastShadow = true
	end)

	-- strip old welds so we can rebind cleanly
	for _, __c in ipairs(__handle:GetChildren()) do
		if __c:IsA('Weld') or __c:IsA('WeldConstraint') or __c.Name == 'AccessoryWeld' then
			__c:Destroy()
		end
	end
end

local function __weld_accessory(__char: Model, __acc: Accessory | Hat): boolean
	local __handle = __acc:FindFirstChild('Handle')

	if not __handle or not __handle:IsA('BasePart') then
		-- still parent; better than drop
		__acc.Parent = __char

		return __acc.Parent == __char
	end

	__prepare_handle(__handle)

	local __hum = __humanoid(__char)

	-- 1) native AddAccessory (best when it works)
	if __hum then
		__acc.Parent = game -- must not already be under character for some clients
		local __ok = pcall(function()
			__hum:AddAccessory(__acc)
		end)

		if __ok and __acc.Parent == __char then
			local __weld = __handle:FindFirstChild('AccessoryWeld') or __handle:FindFirstChildOfClass('Weld')

			if __weld then
				return true
			end
		end
	end

	-- 2) attachment-based weld (R6 / R15 rigid)
	__acc.Parent = __char

	local __acc_att: Attachment? = nil

	for _, __c in ipairs(__handle:GetChildren()) do
		if __c:IsA('Attachment') then
			__acc_att = __c
			break
		end
	end

	if not __acc_att then
		for _, __c in ipairs(__acc:GetDescendants()) do
			if __c:IsA('Attachment') and __c.Parent == __handle then
				__acc_att = __c
				break
			end
		end
	end

	local __body_att: Attachment? = nil

	if __acc_att then
		__body_att = __find_attachment(__char, __acc_att.Name)
	end

	-- common fallbacks by accessory type
	if not __body_att then
		local __atype = ''

		pcall(function()
			if __acc:IsA('Accessory') then
				__atype = tostring(__acc.AccessoryType)
			end
		end)

		local __prefer = {
			'HatAttachment',
			'HairAttachment',
			'FaceFrontAttachment',
			'FaceCenterAttachment',
			'NeckAttachment',
			'NeckRigAttachment',
			'LeftCollarAttachment',
			'RightCollarAttachment',
			'BodyFrontAttachment',
			'BodyBackAttachment',
			'WaistFrontAttachment',
			'WaistCenterAttachment',
			'WaistBackAttachment',
			'FaceAccessoryAttachment',
		}

		if string.find(__atype, 'Hair') then
			__prefer = { 'HairAttachment', 'HatAttachment', 'FaceCenterAttachment' }
		elseif string.find(__atype, 'Face') then
			__prefer = { 'FaceFrontAttachment', 'FaceCenterAttachment', 'HatAttachment' }
		elseif string.find(__atype, 'Back') then
			__prefer = { 'BodyBackAttachment', 'WaistBackAttachment' }
		elseif string.find(__atype, 'Waist') then
			__prefer = { 'WaistCenterAttachment', 'WaistFrontAttachment', 'BodyFrontAttachment' }
		elseif string.find(__atype, 'Neck') then
			__prefer = { 'NeckAttachment', 'NeckRigAttachment' }
		end

		for _, __name in ipairs(__prefer) do
			__body_att = __find_attachment(__char, __name)

			if __body_att then
				break
			end
		end
	end

	if not __body_att then
		local __head = __char:FindFirstChild('Head')

		if __head then
			__body_att = __head:FindFirstChild('HatAttachment') :: Attachment?
				or __head:FindFirstChild('HairAttachment') :: Attachment?
				or __head:FindFirstChildOfClass('Attachment')
		end
	end

	if __body_att and __body_att.Parent and __body_att.Parent:IsA('BasePart') then
		local __weld = Instance.new('Weld')
		__weld.Name = 'AccessoryWeld'
		__weld.Part0 = __body_att.Parent
		__weld.Part1 = __handle

		if __acc_att then
			__weld.C0 = __body_att.CFrame
			__weld.C1 = __acc_att.CFrame
		else
			__weld.C0 = CFrame.new(0, 0.5, 0)
			__weld.C1 = CFrame.new()
		end

		__weld.Parent = __handle

		return true
	end

	-- 3) last resort: weld to Head
	local __head = __char:FindFirstChild('Head')

	if __head and __head:IsA('BasePart') then
		local __weld = Instance.new('Weld')
		__weld.Name = 'AccessoryWeld'
		__weld.Part0 = __head
		__weld.Part1 = __handle
		__weld.C0 = CFrame.new(0, 0.6, 0)
		__weld.C1 = CFrame.new()
		__weld.Parent = __handle

		return true
	end

	return __acc.Parent == __char
end

local function __extract_accessories(__root: Instance): { Accessory | Hat }
	local __list: { Accessory | Hat } = {}

	if __root:IsA('Accessory') or __root:IsA('Hat') then
		table.insert(__list, __root :: any)

		return __list
	end

	for _, __c in ipairs(__root:GetChildren()) do
		if __c:IsA('Accessory') or __c:IsA('Hat') then
			table.insert(__list, __c :: any)
		end
	end

	if #__list == 0 then
		for _, __c in ipairs(__root:GetDescendants()) do
			if __c:IsA('Accessory') or __c:IsA('Hat') then
				table.insert(__list, __c :: any)
			end
		end
	end

	return __list
end

local function __attach_accessory_instance(__char: Model, __acc: Accessory | Hat, __seen_names: { [string]: boolean }): boolean
	local __name = __acc.Name

	if __seen_names[__name] then
		-- allow duplicate names with suffix
		local __i = 2

		while __seen_names[__name .. '_' .. tostring(__i)] do
			__i += 1
		end

		__name = __name .. '_' .. tostring(__i)
		__acc.Name = __name
	end

	local __clone = __acc:Clone()
	__clone.Name = __name

	local __ok = __weld_accessory(__char, __clone :: any)

	if __ok then
		__seen_names[__name] = true
	else
		pcall(function()
			__clone:Destroy()
		end)
	end

	return __ok
end

local function __load_accessories_by_ids(__char: Model, __ids: { number }, __seen: { [string]: boolean }): number
	local __n = 0

	for _, __id in ipairs(__ids) do
		local __objs = __get_objects(__id)

		for _, __inst in ipairs(__objs) do
			for _, __acc in ipairs(__extract_accessories(__inst)) do
				if __attach_accessory_instance(__char, __acc, __seen) then
					__n += 1
				end
			end
		end
	end

	return __n
end

local function __clear_visuals(__char: Model): ()
	for _, __c in ipairs(__char:GetChildren()) do
		if __c:IsA('Accessory')
			or __c:IsA('Hat')
			or __c:IsA('Shirt')
			or __c:IsA('Pants')
			or __c:IsA('ShirtGraphic')
			or __c:IsA('BodyColors')
			or __c:IsA('CharacterMesh')
		then
			__c:Destroy()
		end
	end

	local __head = __char:FindFirstChild('Head')

	if __head then
		for _, __d in ipairs(__head:GetChildren()) do
			if __d:IsA('Decal') then
				__d:Destroy()
			end
		end
	end
end

local function __apply_clothing_and_body(__char: Model, __desc: HumanoidDescription): number

	if __desc.Shirt and __desc.Shirt > 0 then

		__shirt.Name = 'Shirt'
		__shirt.ShirtTemplate = __asset_uri(__desc.Shirt)
		__shirt.Parent = __char

	end

	if __desc.Pants and __desc.Pants > 0 then
		local __pants = Instance.new('Pants')
		__pants.Name = 'Pants'
		__pants.PantsTemplate = __asset_uri(__desc.Pants)
		__pants.Parent = __char
		__n += 1
	end

	if __desc.GraphicTShirt and __desc.GraphicTShirt > 0 then
		local __g = Instance.new('ShirtGraphic')
		__g.Name = 'Shirt Graphic'
		__g.Graphic = __asset_uri(__desc.GraphicTShirt)
		__g.Parent = __char
		__n += 1
	end

	local __bc = Instance.new('BodyColors')
	__bc.Name = 'Body Colors'
	__bc.HeadColor3 = __desc.HeadColor
	__bc.TorsoColor3 = __desc.TorsoColor
	__bc.LeftArmColor3 = __desc.LeftArmColor
	__bc.RightArmColor3 = __desc.RightArmColor
	__bc.LeftLegColor3 = __desc.LeftLegColor
	__bc.RightLegColor3 = __desc.RightLegColor
	__bc.Parent = __char
	__n += 1

	local __head = __char:FindFirstChild('Head')

	if __head and __desc.Face and __desc.Face > 0 then
		local __face = Instance.new('Decal')
		__face.Name = 'face'
		__face.Face = Enum.NormalId.Front
		__face.Texture = __asset_uri(__desc.Face)
		__face.Parent = __head

	elseif __head then
		-- keep a default face decal if description has 0
		local __face = Instance.new('Decal')
		__face.Name = 'face'
		__face.Face = Enum.NormalId.Front
		__face.Texture = 'rbxasset://textures/face.png'
		__face.Parent = __head
	end

	return __n
end

local function __apply_packages(__char: Model, __desc: HumanoidDescription): number
	local __n = 0
	local __packages = {
		__desc.Head,
		__desc.Torso,
		__desc.LeftArm,
		__desc.RightArm,
		__desc.LeftLeg,
		__desc.RightLeg,
	}

	for _, __pid in ipairs(__packages) do
		if type(__pid) == 'number' and __pid > 0 then
			for _, __inst in ipairs(__get_objects(__pid)) do
				if __inst:IsA('CharacterMesh') then
					__inst.Parent = __char
					__n += 1
				else
					for _, __ch in ipairs(__inst:GetDescendants()) do
						if __ch:IsA('CharacterMesh') then
							__ch:Clone().Parent = __char
							__n += 1
						end
					end
				end
			end
		end
	end

	return __n
end

local function __copy_base_from_model(__char: Model, __model: Model): number
	local __n = 0
	local __head = __char:FindFirstChild('Head')

	for _, __c in ipairs(__model:GetChildren()) do
		if __c:IsA('Shirt')
			or __c:IsA('Pants')
			or __c:IsA('ShirtGraphic')
			or __c:IsA('BodyColors')
			or __c:IsA('CharacterMesh')
		then
			__c:Clone().Parent = __char
			__n += 1
		elseif __c.Name == 'Head' and __head then
			local __face = __c:FindFirstChild('face') or __c:FindFirstChildOfClass('Decal')

			if __face and __face:IsA('Decal') then
				__face:Clone().Parent = __head
				__n += 1
			end
		end
	end

	return __n
end

local function __copy_accessories_from_model(__char: Model, __model: Model, __seen: { [string]: boolean }): number
	local __n = 0

	for _, __acc in ipairs(__extract_accessories(__model)) do
		if __attach_accessory_instance(__char, __acc, __seen) then
			__n += 1
		end
	end

	return __n
end

local function __copy_accessories_from_character(__char: Model, __source: Model, __seen: { [string]: boolean }): number
	local __n = 0

	for _, __acc in ipairs(__extract_accessories(__source)) do
		if __attach_accessory_instance(__char, __acc, __seen) then
			__n += 1
		end
	end

	return __n
end

local function __apply_appearance(__desc: HumanoidDescription, __live_char: Model?): (boolean, string)
	local __char = __wait_character(LocalPlayer, 5)

	if not __char then
		return false, 'no local character'
	end

	local __hum = __humanoid(__char)

	if not __hum then
		return false, 'no local humanoid'
	end

	local __rig = __hum.RigType
	local __work_desc = __clone_desc(__desc) or __desc
	local __seen_acc: { [string]: boolean } = {}
	local __stats = {
		base = 0,
		acc_live = 0,
		acc_model = 0,
		acc_ids = 0,
		packages = 0,
	}

	-- build reference model (same rig) for clothing / packages / some accs
	local __model: Model? = nil
	local __ok_model, __built = pcall(function()
		return Players:CreateHumanoidModelFromDescription(__work_desc, __rig)
	end)

	if __ok_model and typeof(__built) == 'Instance' and __built:IsA('Model') then
		__model = __built
	end

	__clear_visuals(__char)

	-- clothing / body / face / packages from model first
	if __model then
		__stats.base = __copy_base_from_model(__char, __model)
	end

	-- always reinforce clothing from description (model can miss graphic tee etc.)
	if not __char:FindFirstChildOfClass('Shirt') or not __char:FindFirstChildOfClass('Pants') or not __char:FindFirstChildOfClass('BodyColors') then
		__stats.base += __apply_clothing_and_body(__char, __work_desc)
	else
		-- still ensure graphic tee / face if missing
		if __work_desc.GraphicTShirt and __work_desc.GraphicTShirt > 0 and not __char:FindFirstChildOfClass('ShirtGraphic') then
			local __g = Instance.new('ShirtGraphic')
			__g.Name = 'Shirt Graphic'
			__g.Graphic = __asset_uri(__work_desc.GraphicTShirt)
			__g.Parent = __char
			__stats.base += 1
		end
	end

	if not __char:FindFirstChildWhichIsA('CharacterMesh') then
		__stats.packages = __apply_packages(__char, __work_desc)
	end

	-- ACCESSORIES — always multi-source (this was the bug: skipped when clothing count >= 3)
	-- 1) live character in server (most accurate)
	if __live_char then
		__stats.acc_live = __copy_accessories_from_character(__char, __live_char, __seen_acc)
	end

	-- 2) temp model accessories
	if __model then
		__stats.acc_model = __copy_accessories_from_model(__char, __model, __seen_acc)
	end

	-- 3) force-load every accessory asset id from description
	local __acc_ids = __collect_accessory_ids(__work_desc)
	__stats.acc_ids = __load_accessories_by_ids(__char, __acc_ids, __seen_acc)

	if __model then
		pcall(function()
			__model:Destroy()
		end)
	end

	local __acc_total = __stats.acc_live + __stats.acc_model + __stats.acc_ids
	local __total = __stats.base + __stats.packages + __acc_total

	if __total <= 0 then
		return false, 'could not apply visuals on client'
	end

	return true, string.format(
		'applied · acc %d (live %d / model %d / assets %d) · base %d',
		__acc_total,
		__stats.acc_live,
		__stats.acc_model,
		__stats.acc_ids,
		__stats.base + __stats.packages
	)
end

local function __snapshot_local(): (HumanoidDescription?, string?)
	local __char = __wait_character(LocalPlayer, 5)
	local __hum = __humanoid(__char)

	if __hum then
		local __ok, __desc = pcall(function()
			return __hum:GetAppliedDescription()
		end)

		if __ok and __desc and __desc:IsA('HumanoidDescription') then
			return __clone_desc(__desc) or __desc, nil
		end
	end

	local __ok2, __desc2 = pcall(function()
		return Players:GetHumanoidDescriptionFromUserId(LocalPlayer.UserId)
	end)

	if __ok2 and __desc2 and __desc2:IsA('HumanoidDescription') then
		return __clone_desc(__desc2) or __desc2, nil
	end

	return nil, 'could not snapshot self'
end

local function __player_names(): { string }
	local __out = {}

	for _, __p in ipairs(Players:GetPlayers()) do
		if __p ~= LocalPlayer then
			table.insert(__out, __p.Name)
		end
	end

	table.sort(__out)

	return __out
end

function AvatarStealer:init(): AvatarStealer
	self.__target = ''
	self.__busy = false
	self.__library = nil

	if not self.__saved then
		task.defer(function()
			self:save_self()
		end)
	end

	return self
end

function AvatarStealer:save_self(): (boolean, string)
	local __desc, __err = __snapshot_local()

	if not __desc then
		return false, __err or 'save failed'
	end

	self.__saved = __desc

	return true, 'saved'
end

function AvatarStealer:steal(__target: string | number | Player): (boolean, string)
	if self.__busy then
		return false, 'busy'
	end

	self.__busy = true

	local __ok_run, __result_ok, __result_msg = pcall(function()
		if not self.__saved then
			self:save_self()
		end

		local __label, __user_id, __player = __parse_query(__target)

		if not __user_id then
			return false, 'user not found · use username / userId / profile URL'
		end

		local __desc: HumanoidDescription? = nil
		local __err: string? = nil
		local __live_char: Model? = nil
		local __source = 'catalog'

		if __player then
			__live_char = __wait_character(__player, 2)
			__desc = __desc_from_live(__player)

			if __desc then
				__source = 'live'
			end
		end

		if not __desc then
			__desc, __err = __desc_from_user_id(__user_id)
			__source = 'catalog'
		end

		if not __desc then
			return false, __err or 'no description'
		end

		local __applied, __apply_msg = __apply_appearance(__desc, __live_char)

		if not __applied then
			return false, __apply_msg
		end

		self.__target = __label or tostring(__user_id)

		local __where = if __player then 'in-server' else 'offline'

		return true, 'stole · ' .. self.__target .. ' · ' .. __where .. ' · ' .. __source .. ' · ' .. __apply_msg
	end)

	self.__busy = false

	if not __ok_run then
		return false, tostring(__result_ok)
	end

	return __result_ok :: boolean, __result_msg :: string
end

function AvatarStealer:restore(): (boolean, string)
	if self.__busy then
		return false, 'busy'
	end

	if not self.__saved then
		local __ok_save, __msg_save = self:save_self()

		if not __ok_save or not self.__saved then
			return false, __msg_save or 'nothing saved'
		end
	end

	self.__busy = true

	local __clone = __clone_desc(self.__saved :: HumanoidDescription) or self.__saved
	local __ok, __msg = __apply_appearance(__clone :: HumanoidDescription, nil)

	self.__busy = false

	if __ok then
		self.__target = ''
	end

	return __ok, if __ok then 'restored · ' .. __msg else __msg
end

function AvatarStealer:start(): ()
end

function AvatarStealer:stop(): ()
	self.__busy = false
end

	self.__library = __library

				__self.__target = __list[1]
				__name_box_value = __list[1]
			end

			local __count = 0

			for _, __n in ipairs(__list) do
				if __n ~= '(no players)' then
					__count += 1

	__tab:create_button({
		title = 'Steal selected player',
		callback = function()

				__notify(__self, 'pick a player first', 'warn')

				return
			end

			local __ok, __msg = __self:steal(__t)

			__notify(__self, __msg, if __ok then 'normal' else 'error')
		end,
	})

	__tab:create_divider('Self')

	__tab:create_button({
		title = 'Save my avatar',
		callback = function()
			local __ok, __msg = __self:save_self()

			__notify(__self, __msg, if __ok then 'normal' else 'error')
		end,
	})

	__tab:create_button({
		title = 'Restore my avatar',
		callback = function()

	if __old then
		__old:Destroy()
	end

	pcall(function()
		for _, __c in ipairs(workspace.Terrain:GetChildren()) do
			if __c.Name == 'EnrithyReachViz' or __c.Name == '__enrithy_reach_viz' then
				__c:Destroy()
			end
		end
	end)
end

local function __boot()
	__clean_old()

	if type(Library) ~= 'table' or type(Library.init) ~= 'function' then
		error('[enrithy] Library missing — run build_enrithy.ps1')
	end

	if type(Console) == 'table' and type(Console.init) == 'function' then
		task.spawn(function()
			Console:init({ brand = 'enrithy', delay = 0.03 })
			Console:run()
		end)
	end

	Library:init('enrithy')

	if type(EnrithyReach) == 'table' then
		EnrithyReach:init()
		EnrithyReach:attach_ui(Library)
		EnrithyReach:start()
	end

	if type(PingChanger) == 'table' then
		PingChanger:init()
		PingChanger:set_enabled(false)
		PingChanger:attach_ui(Library)
		PingChanger:start()
	end

	if type(AvatarStealer) == 'table' then
		AvatarStealer:init()
		AvatarStealer:attach_ui(Library)
		AvatarStealer:start()
	end

	if type(Config) == 'table' then
		Config:init()
		Config:attach_ui(Library)
	end

	Library:notify({
		title = 'enrithy',
		content = 'loaded',
		duration = 3,
		notify_type = 'normal',
	})

	print('[enrithy] boot OK')
end

local __ok, __err = pcall(__boot)

if not __ok then
	warn('[enrithy] boot ERROR: ' .. tostring(__err))
end
